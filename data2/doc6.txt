SELECT FNAME, LNAME
FROM EMPLOYEE, DEPENDENT
WHERE SSN = DEPSSN AND DEPENDENT.RELATIONSHIP = 'SON';
SELECT FNAME, LNAME
FROM EMPLOYEE
WHERE SSN IN (SELECT DEPSSN
 FROM DEPENDENT
 WHERE RELATIONSHIP = 'SON');
SELECT FNAME, LNAME
FROM EMPLOYEE
WHERE EXISTS (SELECT *
 FROM DEPENDENT
 WHERE SSN = DEPSSN
 AND RELATIONSHIP = 'SON');
